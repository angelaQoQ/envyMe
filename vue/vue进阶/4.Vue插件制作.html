# 4.Vue插件制作

官方推荐插件社区

# 4.1 基本使用:

  1. 在new Vue() 之前调用Vue.use()加载.
  2. 多次调用Vue.use() 只生效一次
main.js中
    // 像使用vue-router一样简单
    1.//引入你的插件文件
        import MyPlugin from '@/utils/myUtil'
    2.//在newVue之前加载
        Vue.use(MyPlugin)
    ....
    // 实例化
    new Vue({...})

# 4.2 制作插件

  1. 插件文件导出对象上包含install方法
  2. install方法接收参数: 参数1:Vue构造器 ; 参数2:配置对象
  3. 在install方法中添加全局的方法,属性等工具
MyPlugin.js
    import Vue from 'vue'

    2.// 准备插件对象 
    MyPlugin = {
        3.// 必须包含install方法
        install( Vue , optionObj  ){ 3.// 参数1 Vue  ;  参数2: 配置对象
            4.//函数内部设置插件功能

            4.1 // 增加全局方法
            Vue.fn = function(){ console.log('fn方法被调用了') }

            4.2 // 增加全局指令
            Vue.derictive('myplugin-directive' , {
                bind(  el , binding , vdom , old_vdom  )  {}
            })

            4.3 // 增加全局混入
            Vue.mixin({
                created(){ /*获取token...*/ }
            })

            4.4 // 添加实例方法
            Vue.prototype.$http = axios插件对象
        }
    }

    1.// 导出插件对象
    export default MyPlugin
