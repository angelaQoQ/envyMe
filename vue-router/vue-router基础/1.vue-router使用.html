# 1.vue-router使用

Vue Router 是Vue官方推荐的路由管理工具,使用它可以方便的维护组件和路由切换维护
使用Vue Router 和 Vue时 只需关心 "组件和路由" 的映射关系

目标布局

html

挖坑,埋线
    1. home.vue中设置基本排版
    <template>
        <div class="home">
        <!-- 拼接首页结构 : 顶部导航Nav + 左侧SlideBars + 右侧主体区域Main-->

            <nav></nav>
        <!-- SideBar 显示主要导航锚点 -->
            <slide-bars></slide-bars>
        <!-- Main中 显示 组件 -->
            <router-view></router-view>
        </div>
    </template>

    2. slide-bars.vue中设置锚点
    <template>
        <div class="slideBars">
            <!-- 根据权限列表设置锚点 -->
            <router-link v-for="..." :to="权限列表项"></router-link>
        </div>
    </template>

        将侧边栏SideBar中的锚点对应组件渲染到主体main中

javascript

router.js 中
    // 创建 router.js , vue-cli脚手架初始化项目会自动创建

    1. // 引入vue-router.js
    import VueRouter from 'vue-router'

    2. // 引入组件对象,在3中映射对象中使用
    import MyComponent from '@/components/MyComponent'

    3. // 创建路由实例并导出
    export default {
        routes:[
            3.1 // 配置路由和组件的映射关系
            {
                name:'login',   // 添加name属性方便后期维护 路由对象
                path:'/login',  // 锚点值
                component:MyComponent,  // 当前锚点生效时加载的组件--------->2代码写这就是 路由懒加载

                children:[],     // 当前锚点值下的子路由

                redirect:'/黑人问号',   // 设置重定向
                alias:'尼克杨`sWTF?'  // 别名, 访问'/黑人问号'时,地址栏里展示'尼克杨`sWTF'
                props:true,      // 路由参数this.$route.params会被设置成为当前组件的变量
                
                meta:{          // 当前路由映射对象的元数据
                    keepAlive:true      // 设置组件缓存,配合<router-link v-if="">使用
                },
            },
            3.2 ... //更多的路由对象...
        ]
    }
