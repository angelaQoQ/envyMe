# 3.编程式导航

# 3.1编程导航方方法

  1. this.$router.push()

    • 跳转到不同的url，但这个方法回向history栈添加一个记录，点击后退会返回到上一个页面。

  2. this.$router.replace()

    • 同样是跳转到指定的url，但是这个方法不会向history里面添加新的记录

  3. this.$router.go(n)

    • 相对于当前页面向前或向后跳转多少个页面,类似 window.history.go(n)

export default{
    data(){
        return {
            userList:[ /*后台用户数据列表*/ ]
        }
    },
    methods:{
        go2Edit(id){
// 操作$router
            1.// 直接跳转
            this.$router.push( 'edit' )

            2.//通过路由对象名
            this.$router.push( {name:'edit'} )

            3.1// params形式传参    '/eidt/1'
            let url = `/eidt/${id}`
            this.$router.push( url )
            this.$router.push({ path: url })
            this.$router.push({ name:'edit' , params:{ id } }) 

            3.2// query形式传参     'edit?id=1'
            this.$router.push({ path:'eidt' , query:{ id } })
        }
    }
}
