    # 4.Vuex中辅助函数

    # 目的:为了减少使用Vuex时在组件中写过多的调用代码,vuex准备了:

    1. mapState
    2. mapGetters
    3. mapMutations
    4. mapActions

    # 使用

    • mapState和mapGetters在computed中引入
    • mapAction和mapMutation在methods中引入

    准备仓库

    store.js中
export default new Vuex.Store({
    state:{
        aaa:'qwe'
    },
    getters:{
        AAA(state){
            return String.toUpperCase(state.aaa)
        }
    },
    mutations:{
        setAAA(state, newVal){
            state.aaa = newVal
        }
    },
    actions:{
        delaySetAAA(context, newVal){
            setTimeOut(()=>{
                context.commit( 'setA'  , newVal)
            })
        }
    }
})

    组件中使用

    组件内使用
    import {mapState , mapGetters} from 'vuex'
    import {mapMutations , mapActions} from 'vuex'
    export default{
        // 获取数据
        computed:{
            ---1.使用mapState mapGetters---------------------------------
            ...mapState({
                aaa:'aaa'
            }),
            ...mapGetters({
                AAA:'AAA'
            }),
            // ---2.不使用mapState  mapGetters---------------------------------
            // aaa(){
            //     return this.$store.state.aaa
            // },
            // AAA(){
            //     return this.$store.getters.AAA
            // }
        },

        // 修改数据
        methods:{
            ---1.使用mapMutations mapActions---------------------------------
            ...mapMutations(["setAAA"])
            ...mapActions(["delaySetAAA"])
            // 在事件处理函数中直接使用
            eventHandler(){
                this.setAAA(1)
                this.delaySetAAA(11)
            }
            // ---2.不使用用 mapMutations maActions---------------------------------
            // // 在事件处理函数中使用
            // eventHandler(){
            //     this.$store.commit('setAAA' , 1)
            //     this.$store.dispatch('delaySetAAA' , 11)
            // }
        }

        mounted(){
            console.log(this.aaa , this.AAA)
        }
    }
