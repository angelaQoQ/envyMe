# 3.模块化状态管理

# 3.1 必要性

  • 很多状态集中在一个state对象中不利于维护和阅读.

# 3.2 步骤

    modules:{
        bikes,
        cars

})

# 2. 设置数据模块----cars/index.js

cars.js
export default{
  // 使用当前模块名作区分,就是所在文件夹名称
    namespaced: true,

    state:{
        nameList:['奇瑞','红旗','三蹦子','杀马特','狮子头']
    },
    getters:{
        handleNameList(state){}
    },
    mutations:{
        setNameList(state , newList){}
    },
    actions:{
        delaySetNameList(context , newList){
            context.commit('setNameList' , newList )
        }
    }
}

# 3. 组件中使用状态管理

组件中

  import { mapState , mapMutations , mapActions } from 'vuex'

    export default {
        name:'com',
        // 获取state值
        computed:{
            ...mapState('cars/index.js' ,{ //第一个参数模块名, 文件夹层及路径写清楚
                nameList : state => state.nameList
            } )
        },
        // 触发actions
        methods:{
            ...mapActions('cars/index.js' , []) // 第一个参数模块名
        }
    }
